    function scaleCoordinates(drawing, b1, b2, k) {
  let scaledDrawing = [];
  for (let i = 0; i < drawing.length; i++) {
    let scaledX = (drawing[i].x+b1) * k;
    let scaledY = (drawing[i].y+b2) * k;
    scaledDrawing.push({'x': scaledX, 'y': scaledY});
    }
  return scaledDrawing;
}
